// ========== LIBRERÍAS ==========
#include <WiFi.h>
#include <WebServer.h>
#include <DHT.h>
#include <ArduinoJson.h>

// ========== CONFIGURACIÓN WiFi ==========
const char* ssid = "TU_NOMBRE_WIFI";           // 👈 CAMBIA ESTO
const char* password = "TU_CONTRASEÑA_WIFI";   // 👈 CAMBIA ESTO

// ========== PINES DE HARDWARE ==========
#define DHTPIN 4              // GPIO4 -> Sensor DHT11 (DATA)
#define DHTTYPE DHT11         // Tipo de sensor
#define SOIL_SENSOR_PIN 34    // GPIO34 -> Sensor Humedad Suelo (A0)
#define RELAY_PIN 26          // GPIO26 -> Módulo Relé (IN1)

// ========== CONFIGURACIÓN DEL SISTEMA ==========
const int UMBRAL_SECO = 30;           // Porcentaje mínimo para activar riego
const int TIEMPO_RIEGO = 3000;        // Milisegundos de riego (3 segundos)
const unsigned long INTERVALO_LECTURA = 2000;  // Leer sensores cada 2 segundos

// Lógica de relé (Muchos módulos son Active Low)
const int BOMBA_ENCENDIDA = LOW;      
const int BOMBA_APAGADA = HIGH;

// ========== OBJETOS GLOBALES ==========
DHT dht(DHTPIN, DHTTYPE);
WebServer server(80);

// ========== VARIABLES DE ESTADO ==========
struct SensorData {
    float temperatura;
    float humedad;
    int humedadSuelo;
    unsigned long ultimaLectura;
} sensores;

struct SystemState {
    bool bombaActiva;
    String modo;  // "auto" o "manual"
    unsigned long tiempoInicioRiego;
    bool rigoEnProceso;
} sistema;

// ========================================
// SETUP - INICIALIZACIÓN
// ========================================
void setup() {
    Serial.begin(115200);
    delay(1000);
    
    Serial.println("\n\n========================================");
    Serial.println("🌱 INVERNADERO INTELIGENTE - ESP32");
    Serial.println("========================================\n");
    
    // Configurar pines
    pinMode(RELAY_PIN, OUTPUT);
    digitalWrite(RELAY_PIN, BOMBA_APAGADA);  // Asegurar que inicie apagada
    
    // Inicializar sensor DHT11
    dht.begin();
    Serial.println("✓ Sensor DHT11 inicializado");
    
    // Estado inicial del sistema
    sistema.bombaActiva = false;
    sistema.modo = "manual"; // 👈 Cambiado a manual por defecto
    sistema.rigoEnProceso = false;
    sistema.tiempoInicioRiego = 0;
    
    // Conectar a WiFi
    conectarWiFi();
    
    // Configurar rutas del servidor web
    configurarServidor();
    
    // Iniciar servidor
    server.begin();
    Serial.println("✓ Servidor web iniciado");
    Serial.println("\n========================================");
    Serial.println("📱 DASHBOARD LISTO");
    Serial.println("========================================");
    Serial.print("🌐 Accede desde tu móvil a: http://");
    Serial.println(WiFi.localIP());
    Serial.println("========================================\n");
}

// ========================================
// LOOP PRINCIPAL
// ========================================
void loop() {
    // Manejar peticiones HTTP
    server.handleClient();
    
    // Leer sensores cada 2 segundos
    if (millis() - sensores.ultimaLectura >= INTERVALO_LECTURA) {
        leerSensores();
        sensores.ultimaLectura = millis();
        
        // Mostrar datos en Serial
        imprimirDatos();
    }
    
    // Lógica de riego automático
    if (sistema.modo == "auto") {
        controlarRiegoAutomatico();
    }
}

// ========================================
// FUNCIÓN: CONECTAR WiFi
// ========================================
void conectarWiFi() {
    Serial.print("📡 Conectando a WiFi: ");
    Serial.println(ssid);
    
    WiFi.begin(ssid, password);
    
    int intentos = 0;
    while (WiFi.status() != WL_CONNECTED && intentos < 30) {
        delay(500);
        Serial.print(".");
        intentos++;
    }
    
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("\n✓ WiFi conectado exitosamente!");
        Serial.print("📍 Dirección IP: ");
        Serial.println(WiFi.localIP());
        Serial.print("📶 Intensidad señal: ");
        Serial.print(WiFi.RSSI());
        Serial.println(" dBm");
    } else {
        Serial.println("\n❌ Error: No se pudo conectar a WiFi");
        Serial.println("Verifica el nombre y contraseña de tu red");
    }
}

// ========================================
// FUNCIÓN: LEER SENSORES
// ========================================
void leerSensores() {
    // Leer DHT11 (Temperatura y Humedad Ambiental)
    sensores.temperatura = dht.readTemperature();
    sensores.humedad = dht.readHumidity();
    
    // Verificar si la lectura es válida
    if (isnan(sensores.temperatura) || isnan(sensores.humedad)) {
        Serial.println("⚠️ Error leyendo sensor DHT11");
        sensores.temperatura = 0;
        sensores.humedad = 0;
    }
    
    // Leer sensor de humedad del suelo (analógico)
    int valorAnalogico = analogRead(SOIL_SENSOR_PIN);
    
    // Convertir a porcentaje (0-4095 -> 0-100%)
    // NOTA: Calibra estos valores según tu sensor
    // Valores típicos: Seco=3500-4095, Húmedo=1000-1500
    sensores.humedadSuelo = map(valorAnalogico, 4095, 1000, 0, 100);
    sensores.humedadSuelo = constrain(sensores.humedadSuelo, 0, 100);
}

// ========================================
// FUNCIÓN: CONTROL AUTOMÁTICO DE RIEGO
// ========================================
void controlarRiegoAutomatico() {
    // Si el suelo está seco y la bomba no está activa, encender
    if (sensores.humedadSuelo < UMBRAL_SECO && !sistema.bombaActiva) {
        Serial.println("\n🚨 ALERTA: Suelo seco detectado. Activando riego...");
        encenderBomba();
    } 
    // Si el suelo ya está húmedo (umbral + 10 para evitar encendidos constantes), apagar
    else if (sensores.humedadSuelo > (UMBRAL_SECO + 10) && sistema.bombaActiva) {
        Serial.println("\n✅ INFO: Humedad suficiente. Deteniendo riego.");
        apagarBomba();
    }
}

// ========================================
// FUNCIONES: CONTROL DE BOMBA
// ========================================
void encenderBomba() {
    digitalWrite(RELAY_PIN, BOMBA_ENCENDIDA);  // Activar relé
    sistema.bombaActiva = true;
    Serial.println("✓ Bomba ENCENDIDA");
}

void apagarBomba() {
    digitalWrite(RELAY_PIN, BOMBA_APAGADA);   // Desactivar relé
    sistema.bombaActiva = false;
    Serial.println("✓ Bomba APAGADA");
}

// ========================================
// FUNCIÓN: IMPRIMIR DATOS EN SERIAL
// ========================================
void imprimirDatos() {
    Serial.println("\n--- LECTURA DE SENSORES ---");
    Serial.print("🌡️  Temperatura: ");
    Serial.print(sensores.temperatura);
    Serial.println(" °C");
    
    Serial.print("💧 Humedad Aire: ");
    Serial.print(sensores.humedad);
    Serial.println(" %");
    
    Serial.print("🌱 Humedad Suelo: ");
    Serial.print(sensores.humedadSuelo);
    Serial.println(" %");
    
    Serial.print("💦 Bomba: ");
    Serial.println(sistema.bombaActiva ? "ENCENDIDA" : "APAGADA");
    
    Serial.print("🤖 Modo: ");
    Serial.println(sistema.modo);
    Serial.println("---------------------------");
}

// ========================================
// CONFIGURACIÓN DEL SERVIDOR WEB
// ========================================
void configurarServidor() {
    // Ruta principal: Servir el dashboard
    server.on("/", HTTP_GET, []() {
        server.send(200, "text/html", paginaHTML());
    });
    
    // Endpoint: Obtener datos de sensores (GET)
    server.on("/data", HTTP_GET, []() {
        StaticJsonDocument<256> doc;
        
        doc["temperature"] = sensores.temperatura;
        doc["humidity"] = sensores.humedad;
        doc["soil"] = sensores.humedadSuelo;
        doc["pumpActive"] = sistema.bombaActiva;
        doc["mode"] = sistema.modo;
        doc["timestamp"] = millis();
        
        String json;
        serializeJson(doc, json);
        
        // Headers CORS para permitir peticiones desde cualquier origen
        server.sendHeader("Access-Control-Allow-Origin", "*");
        server.send(200, "application/json", json);
    });
    
    // Endpoint: Recibir comandos de control (POST)
    server.on("/control", HTTP_POST, []() {
        if (server.hasArg("plain")) {
            String body = server.arg("plain");
            
            StaticJsonDocument<128> doc;
            DeserializationError error = deserializeJson(doc, body);
            
            if (error) {
                server.send(400, "application/json", "{\"error\":\"JSON inválido\"}");
                return;
            }
            
            String command = doc["command"];
            
            // Comando: Cambiar modo
            if (command == "mode") {
                sistema.modo = doc["value"].as<String>();
                Serial.print("📝 Modo cambiado a: ");
                Serial.println(sistema.modo);
                
                server.sendHeader("Access-Control-Allow-Origin", "*");
                server.send(200, "application/json", "{\"status\":\"ok\",\"mode\":\"" + sistema.modo + "\"}");
            }
            // Comando: Control manual de bomba
            else if (command == "pump") {
                if (sistema.modo == "manual") {
                    bool activate = doc["value"];
                    
                    if (activate && !sistema.bombaActiva) {
                        encenderBomba();
                    } else if (!activate && sistema.bombaActiva) {
                        apagarBomba();
                    }
                    
                    server.sendHeader("Access-Control-Allow-Origin", "*");
                    server.send(200, "application/json", "{\"status\":\"ok\",\"pump\":" + String(sistema.bombaActiva ? "true" : "false") + "}");
                } else {
                    server.sendHeader("Access-Control-Allow-Origin", "*");
                    server.send(400, "application/json", "{\"error\":\"Bomba en modo automático\"}");
                }
            }
            else {
                server.sendHeader("Access-Control-Allow-Origin", "*");
                server.send(400, "application/json", "{\"error\":\"Comando desconocido\"}");
            }
        }
    });
    
    // Manejar OPTIONS para CORS
    server.on("/control", HTTP_OPTIONS, []() {
        server.sendHeader("Access-Control-Allow-Origin", "*");
        server.sendHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS");
        server.sendHeader("Access-Control-Allow-Headers", "Content-Type");
        server.send(204);
    });
    
    // Ruta no encontrada
    server.onNotFound([]() {
        server.send(404, "text/plain", "404: Página no encontrada");
    });
}
